#WRITE A PROGRAM IN PYTHON THAT STORES ALPHABETS FROM A TO Z IN A TEXT FILE.
IMPORT STRING
F = OPEN("ANSWER#1.TXT", 'W') #WRITE MODE
A=[]
FOR I IN STRING.ASCII_LOWERCASE:
	A.APPEND(I)
PRINT(A, FILE=F)

F.CLOSE()
